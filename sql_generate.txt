create table admin_user
(
    username varchar(20) null,
    password varchar(30) null
);

create table file
(
    id   int auto_increment
        primary key,
    name varchar(2000) not null,
    path varchar(2000) not null
);

create table item
(
    id      int auto_increment
        primary key,
    price   int          not null,
    type_id int          not null,
    name    varchar(200) null
);

create table item_image
(
    id      int auto_increment
        primary key,
    item_id int not null,
    file_id int not null
);

create table item_type
(
    ITEM_TYPE_ID int auto_increment
        primary key,
    NAME         varchar(100) not null,
    CREATE_USER  varchar(10)  null,
    CREATE_DATE  date         null
);

create table sale_order
(
    id          varchar(200) not null
        primary key,
    user_id     varchar(200) not null,
    create_date datetime     not null,
    item_id     varchar(200) not null,
    status      int          not null,
    account_id  varchar(200) null
);

create table stock_account
(
    id          varchar(200) not null
        primary key,
    create_date datetime     not null,
    create_user varchar(200) null,
    status      int          not null,
    username    varchar(200) not null,
    password    varchar(200) not null,
    column_name int          null,
    item_id     decimal      not null
);

create table top_up
(
    id          int auto_increment
        primary key,
    create_date datetime     not null,
    amount      decimal      not null,
    status      decimal      not null,
    username    varchar(200) not null,
    method      int          not null
);

create table user
(
    USER_ID     varchar(200) not null
        primary key,
    USERNAME    varchar(20)  not null,
    PASSWORD    varchar(100) not null,
    balance     mediumtext   not null,
    create_date int          null
);
